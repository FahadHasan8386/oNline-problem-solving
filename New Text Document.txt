#include <bits/stdc++.h>
using namespace std ;

int main ()
{
    int t ;
    cin >> t;

    while (t--) {
        int n;
        cin >> n;
        int cnt = 0;
        vector<int> s(n);

        for (int i = 0; i < n; i++) {
            cin >> s[i];
        }

        for (int i = 1; i < n; i++) {
            if (s[0] != s[i]) {
                cnt++;
            }
        }

        cout << cnt << endl;
    }
}
